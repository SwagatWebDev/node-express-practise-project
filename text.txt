private String getOldValueOnCreationDay(EventLogProjection event) {
    // Convert String to LocalDateTime
    LocalDateTime creationDateTime = LocalDateTime.parse(event.getEventCreatedDate(), DateTimeFormatter.ISO_DATE_TIME);

    // Extract LocalDate
    LocalDate creationDay = creationDateTime.toLocalDate();
    
    // If the creation day is the same, use the original value, otherwise, use an empty string or some default value
    return creationDay.equals(LocalDate.now()) ? event.getOldValue() : "";
}
