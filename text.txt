List<EventLogDto> result = Stream.concat(
        groupedByNameAndPathAndDay.values().stream()
                .flatMap(List::stream)
                .map(this::processGroupedEvents),
        unMatchedPathResults.stream()
                .map(this::processUnGroupedEvents)
)
        .filter(event -> !Objects.equals(event.getOldValue(), event.getNewValue())
                && StringUtils.isNotBlank(event.getNewValue()))
        .sorted(Comparator.comparing(EventLogOto::getEventCreateMate).reversed())
        .collect(Collectors.toList());
