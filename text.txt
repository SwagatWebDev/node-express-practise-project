@Override
public Page<EventLogDto> getEventLog(Pageable pageable) {
    Page<EventLogProjection> eventLogData = eventLogRepository.getEventLogData(pageable);

    // Create a map with a key as a combination of name and date, and value as EventLogProjection
    Map<String, EventLogProjection> uniqueEventsMap = eventLogData.stream()
            .collect(Collectors.toMap(
                    event -> event.getName() + LocalDate.parse(event.getEventCreatedDate()).toString(),
                    Function.identity(),
                    (event1, event2) -> {
                        // Compare dates and pick the one with the lower event created date
                        LocalDate date1 = LocalDate.parse(event1.getEventCreatedDate());
                        LocalDate date2 = LocalDate.parse(event2.getEventCreatedDate());

                        return date1.isBefore(date2) ? event1 : event2;
                    }
            ));

    List<EventLogDto> eventLogDtoList = uniqueEventsMap.values().stream()
            .map(this::mapToEventLogDto)
            .collect(Collectors.toList());

    return new PageImpl<>(eventLogDtoList, pageable, eventLogData.getTotalElements());
}
