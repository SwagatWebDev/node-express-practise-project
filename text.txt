private void updateRecordWithMinOldValueAndMaxNewValue(List<EventLogProjection> eventsWithSameNamePathAndDay) {
    eventsWithSameNamePathAndDay.stream()
            .min(Comparator.comparing(
                    EventLogProjection::getOldValue,
                    Comparator.nullsLast(Comparator.naturalOrder()))
                    .thenComparing(
                            EventLogProjection::getEventCreatedDate,
                            Comparator.nullsLast(Comparator.naturalOrder())))
            .ifPresent(minOldValueEvent -> {
                EventLogProjection maxNewValueEvent = eventsWithSameNamePathAndDay.stream()
                        .filter(event -> Objects.equals(event.getOldValue(), minOldValueEvent.getOldValue()))
                        .max(Comparator.comparing(
                                EventLogProjection::getEventCreatedDate,
                                Comparator.nullsLast(Comparator.naturalOrder())))
                        .orElseThrow();

                // Update the existing record with the new values
                minOldValueEvent.setNewValue(maxNewValueEvent.getNewValue());
                // Update other fields as needed
                minOldValueEvent.setEventCreatedDate(maxNewValueEvent.getEventCreatedDate());
                // Update other fields as needed
            });
}
