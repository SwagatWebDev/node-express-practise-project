Map<String, EventLogProjection> latestRecords = new HashMap<>();

    List<EventLogDto> eventLogDtoList = eventLogData.stream()
            .sorted(Comparator.comparing(this::parseEventCreatedDate).reversed())
            .filter(event -> {
                String key = event.getName();
                LocalDateTime eventDate = parseEventCreatedDate(event);
                return latestRecords.containsKey(key) && eventDate.isAfter(parseEventCreatedDate(latestRecords.get(key)));
            })
            .peek(event -> latestRecords.put(event.getName(), event))
            .map(this::mapToEventLogDto)
            .collect(Collectors.toList());

            private LocalDateTime parseEventCreatedDate(EventLogProjection event) {
                // Assuming your date format is something like "yyyy-MM-dd HH:mm:ss"
                DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss");
                return LocalDateTime.parse(event.getEventCreatedDate(), formatter);
            }
