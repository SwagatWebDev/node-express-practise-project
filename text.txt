private String extractValueFromBusinessCapability(String text) {
    if (text != null) {
        Pattern pattern = Pattern.compile("BusinessCapability : \"(.*?)\"");
        Matcher matcher = pattern.matcher(text);

        String result = matcher.find() ? matcher.group(1) : text;

        // Check if the final text contains "[Deleted Fact Sheet]"
        if (result != null && result.contains("[Deleted Fact Sheet]")) {
            return null;
        }

        return result;
    }
    return text;
}
